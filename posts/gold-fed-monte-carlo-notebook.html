# --- Calculate historical sensitivity of gold to yield changes ---
df['gold_ret']  = np.log(df['gold']  / df['gold'].shift(1))
df['yield_chg'] = df['yield'].diff()   # Daily change in yield (bps-ish)
df = df.dropna()

# Base annualized stats from history
base_mu    = float(df['gold_ret'].mean() * 252)
base_sigma = float(df['gold_ret'].std()  * np.sqrt(252))

# Sensitivity: how much does gold move per 1% drop in yield?
# Using linear regression coefficient
beta = float(np.polyfit(df['yield_chg'], df['gold_ret'], 1)[0])
# Annualize: beta per daily yield change → per 1% annual yield change
# Don't annualize beta — the yield change is a one-time move, not daily
gold_per_100bps = beta  # sensitivity of daily return per daily yield change
   # approx annual gold return per 1% yield move

print(f'Base Annual Drift:              {base_mu*100:.2f}%')
print(f'Base Annual Volatility:         {base_sigma*100:.2f}%')
print(f'Gold sensitivity to yields:     {gold_per_100bps*100:.2f}% per 1% yield change')
print(f'(Negative = gold rises when yields fall, as expected)')

# --- Polymarket Fed Scenarios (Mar-Jun 2026) ---
# Each 25bps cut = -0.25% yield change
scenarios = {
    'Pause-Pause-Pause (67%)': {'prob': 0.67, 'cuts': 0,  'sigma_adj': 0.00},
    'Pause-Pause-Cut   (22%)': {'prob': 0.22, 'cuts': 1,  'sigma_adj': 0.01},
    'Pause-Cut-Cut      (4%)': {'prob': 0.043,'cuts': 2,  'sigma_adj': 0.02},
    'Pause-Cut-Pause    (4%)': {'prob': 0.035,'cuts': 1,  'sigma_adj': 0.01},
}

# Expected yield change = weighted sum of cuts × 0.25%
expected_yield_chg = sum(
    s['prob'] * (-s['cuts'] * 0.25) for s in scenarios.values()
)

# Drift adjustment from Fed path
mu_adj = gold_per_100bps * expected_yield_chg

print('Fed Scenario Probabilities (Polymarket):')
for name, s in scenarios.items():
    print(f'  {name}: {s["cuts"]} cut(s)')
print(f'\nExpected yield change: {expected_yield_chg*100:.2f}%')
print(f'Drift adjustment to gold: {mu_adj*100:+.2f}%')

# --- Run Monte Carlo for each scenario + baseline ---
T           = 1.0
dt          = 1/252
N           = int(T / dt)
simulations = 1000
np.random.seed(42)

# Shared random draws for fair comparison
Z = np.random.standard_normal((N, simulations))

def run_simulation(mu, sigma):
    daily_ret = np.exp((mu - 0.5 * sigma**2) * dt + sigma * np.sqrt(dt) * Z)
    paths = np.zeros((N + 1, simulations))
    paths[0] = S0
    for t in range(1, N + 1):
        paths[t] = paths[t-1] * daily_ret[t-1]
    return paths

results = {}
colors  = {'Baseline (No Fed Adj)': 'gray',
           'Fed-Adjusted (Polymarket)': 'darkorange'}

# Baseline: pure historical
results['Baseline (No Fed Adj)']      = run_simulation(base_mu, base_sigma)
# Fed-adjusted: drift shifted by expected Fed path
results['Fed-Adjusted (Polymarket)']  = run_simulation(base_mu + mu_adj, base_sigma)

# Also run individual scenarios
scenario_colors = ['#2196F3', '#4CAF50', '#FF5722', '#9C27B0']
for (name, s), col in zip(scenarios.items(), scenario_colors):
    adj_mu = base_mu + gold_per_100bps * (-s['cuts'] * 0.25)
    adj_sigma = base_sigma + s['sigma_adj']
    results[name] = run_simulation(adj_mu, adj_sigma)
    colors[name]  = col

print('Simulations complete!')
